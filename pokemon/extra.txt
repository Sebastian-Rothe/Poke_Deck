js

function openPopup(cardElement) {
    const popupContainer = document.getElementById('popup-container');
    const popupContent = document.getElementById('popup-content');

    // Hier kannst du den Inhalt des Pop-ups mit den Details des Pokémon füllen
    const pokemonDetails = getPokemonDetails(cardElement); // Diese Funktion musst du definieren
    fillPopupContent(pokemonDetails); // Diese Funktion musst du definieren

    // Popup anzeigen und Animation starten
    popupContainer.classList.remove('popup-hidden');
    popupContainer.classList.add('popup-visible');
}

function fillPopupContent(details) {
    const popupContent = document.getElementById('popup-content');
    // Füge hier den HTML-Code hinzu, um den Pop-up-Inhalt zu füllen
    popupContent.innerHTML = `
        <h2>${details.name}</h2>
        <img src="${details.image}" alt="${details.name}">
        <p>Type: ${details.types.join(', ')}</p>
        <p>Abilities: ${details.abilities.join(', ')}</p>
        <p>Height: ${details.height}</p>
        <p>Weight: ${details.weight}</p>
        <!-- Weitere Details hier hinzufügen -->
    `;
}

function getPokemonDetails(cardElement) {
    // Diese Funktion muss angepasst werden, um die Details des ausgewählten Pokémon zu erhalten
    // z.B. durch Zugriff auf ein Data-Attribut oder durch eine andere Methode, die du verwendest
    return {
        name: cardElement.querySelector('.pokemon-name').innerText,
        image: cardElement.querySelector('.pokemon-image').src,
        types: ['Type1', 'Type2'], // Ersetze durch echte Daten
        abilities: ['Ability1', 'Ability2'], // Ersetze durch echte Daten
        height: '1.0m', // Ersetze durch echte Daten
        weight: '10kg' // Ersetze durch echte Daten
    };
}
